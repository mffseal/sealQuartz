# 保护性暂停

# 基本实现

Guarded Suspension：用于一个线程等待另一个线程的执行结果，只适用于产生一个消息的情况。

  • 有一个结果需要从一个线程传递到另一个线程，让他们关联同一个 GuardedObject。
  • 如果有结果不断从一个线程到另一个线程那么可以使用消息队列（见生产者/消费者）。
  • JDK 中，join 的实现、Future 的实现，采用的就是此模式。
  • 因为要等待另一方的结果，因此归类到同步模式。

产生结果的线程和消费结果的线程要一一对应。

@Slf4j  
public class Test10GuardedSuspention {  
    public static void main(String[] args) {  
        // T1 等待 T2 下载结果  
        GuardedObj go = new GuardedObj();  
        new Thread(()->{  
            log.debug("等待结果");  
            go.get();  
            log.debug("下载完成");  
        }).start();  
  
        new Thread(()->{  
            log.debug("开始下载");  
            try {  
                go.complate(Downloader.download());  
            } catch (IOException e) {  
                throw new RuntimeException(e);  
            }  
        }).start();  
    }  
}  
  
class GuardedObj {  
    private Object response;  

    // 获取结果，t1调用  
    public Object get(long millis) {  
        synchronized (this) {  
            // 1) 记录最初时间  
            long begin = System.currentTimeMillis();  
            // 2) 已经经历的时间  
            long timePassed = 0;  
  
            // 用循环是因为要判断是否虚假唤醒  
            // 判断东西有没有到
            while (response == null) {  
                // 4) 假设 millis 是 1000，结果在 400 时唤醒了，那么还有 600 要等  
                long waitTime = millis-timePassed;  
                if (waitTime<=0)  
                    break;  
                // 等待结果  
                try {  
                    this.wait(waitTime);  
                } catch (InterruptedException e) {  
                    e.printStackTrace();  
                }  
                // 更新经过的时间  
                timePassed = System.currentTimeMillis()-begin;  
            }  
            return response;  
        }  
    }  
  
    // 产生结果 ，t2调用
    public void complate(Object response) {  
        synchronized (this) {  
            // 给成员变量赋值  
            this.response = response;  
            this.notify();  
        }  
    }  
}  
  
class Downloader {  
    public static List<String> download() throws IOException {  
        List<String> lines = new ArrayList<>();  
        HttpURLConnection conn = (HttpURLConnection) new URL("https://www.baidu.com/").openConnection();  
        BufferedReader reader = new BufferedReader(new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8));  
        String line;  
        while ((line = reader.readLine())!=null)  
            lines.add(line);  
        return lines;  
    }  
}

# 与join方案相比

  • T2完成任务后可以执行其它任务，而join方案下载器下载完成只能等待T1执行完成。
  • join必须借助全局变量传递结果，而本方案结果都是局部的。

package test;  
  
import lombok.extern.slf4j.Slf4j;  
  
import java.io.BufferedReader;  
import java.io.IOException;  
import java.io.InputStreamReader;  
import java.net.HttpURLConnection;  
import java.net.URL;  
import java.nio.charset.StandardCharsets;  
import java.util.*;  
  
@Slf4j  
public class Test10GuardedSuspention {  
    public static void main(String[] args) throws InterruptedException {  
        for (int i=0; i<3; i++) {  
            new People().start();  
        }  
        Thread.sleep(3000);  // 等的时间太短会导致迭代器部分并发错误  
        for (Integer id : MailBoxes.getIds()) {  
            log.debug("叫出邮递员{}", id);  
            new Postman(id, "||内容" + id).start();  
        }  
    }  
}  
  
@Slf4j  
class People extends Thread {  
    @Override  
    public void run() {  
        // 收件人创建邮箱  
        GuardedObj go = MailBoxes.createGuardedObj();  
        log.debug("开始收信{}", go.getId());  
        // 超时等待收信  
        Object mail = go.get(5000);  
        if (mail!=null)  
            log.debug("收到了{},内容{}", go.getId(), mail);  
        else            log.debug("{}超时未收到", go.getId());  
    }  
}  
  
@Slf4j  
class Postman extends Thread {  
    private int id;  
    private String mail;  
  
    public Postman(int id, String mail) {  
        this.id = id;  
        this.mail = mail;  
    }  
  
    @Override  
    public void run() {  
        GuardedObj go = MailBoxes.getGuardedObj(id);  
        log.debug("往{}送信，内容{}", id, mail);  
        go.complate(mail);  
    }  
}  
  
// 类似蜂巢快递柜  
@Slf4j  
class MailBoxes {  
    private static Map<Integer, GuardedObj> box = new Hashtable<>();  // HashTable线程安全，HashMap线程不安全  
    private static int id = 1;  
  
    // 产生唯一ID  
    private static synchronized int generateId() {  
        return id++;  
    }  
  
    public static GuardedObj createGuardedObj() {  
        GuardedObj go = new GuardedObj(generateId());  
        box.put(go.getId(), go);  
        return go;  
    }  
    public static Set<Integer> getIds() {  
        log.debug("{}",box.size());  
        return box.keySet();  
    }  
  
    // 快递柜你取完快递要把格子释放出去  
    public static GuardedObj getGuardedObj(int id) {  
        return box.remove(id);  
    }  
}  
  
class GuardedObj {  
    private Object response;  
    private int id;  
  
    public int getId() {  
        return id;  
    }  
  
    public GuardedObj(int id) {  
        this.id = id;  
    }  
  
    // 获取结果  
    public Object get(long millis) {  
        synchronized (this) {  
            // 1) 记录最初时间  
            long begin = System.currentTimeMillis();  
            // 2) 已经经历的时间  
            long timePassed = 0;  
            // 判断是否虚假唤醒  
            while (response == null) {  
                // 4) 假设 millis 是 1000，结果在 400 时唤醒了，那么还有 600 要等  
                long waitTime = millis-timePassed;  
                if (waitTime<=0)  
                    break;  
                // 等待结果  
                try {  
                    this.wait(waitTime);  
                } catch (InterruptedException e) {  
                    e.printStackTrace();  
                }  
                // 更新经过的时间  
                timePassed = System.currentTimeMillis()-begin;  
            }  
            return response;  
        }  
    }  
  
    // 产生结果  
    public void complate(Object response) {  
        synchronized (this) {  
            // 给成员变量赋值  
            this.response = response;  
            this.notify();  
        }  
    }  
}  
  
class Downloader {  
    public static List<String> download() throws IOException {  
        List<String> lines = new ArrayList<>();  
        HttpURLConnection conn = (HttpURLConnection) new URL("https://www.baidu.com/").openConnection();  
        BufferedReader reader = new BufferedReader(new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8));  
        String line;  
        while ((line = reader.readLine())!=null)  
            lines.add(line);  
        return lines;  
    }  
}
