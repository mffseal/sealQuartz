class Graph {
    private ArrayList<Integer> vertexList; // å­˜æ”¾é¡¶ç‚¹é›†åˆ
    private int[][] edges; // çŸ©é˜µ
    private int num; // è¾¹çš„æ•°ç›®

    public Graph(int n) {
        edges = new int[n][n];
        vertexList = new ArrayList<Integer>(n);
        num = 0;
    }

    // æ’å…¥èŠ‚ç‚¹
    public void addVer(int vertex) {
        vertexList.add(vertex);
    }

    // æ·»åŠ è¾¹
    public void addNum(int v1, int v2, int weight) {
        // v1å’Œv2æŒ‡çš„æ˜¯ç‚¹çš„ä¸‹æ ‡ã€‚weightè¡¨ç¤ºæƒå€¼
        edges[v1][v2] = weight;
        edges[v2][v1] = weight;
        num++;
    }

    // è¿”å›èŠ‚ç‚¹çš„ä¸ªæ•°
    public int nodeNum() {
        return vertexList.size();
    }

    // è¿”å›è¾¹ä¸€å…±æœ‰å¤šå°‘æ¡
    public int edgeNum() {
        return num;
    }

    // é€šè¿‡ç´¢å¼•è¿”å›å€¼
    public Integer getValue(Integer i) {
        return vertexList.get(i);
    }

    // è¿”å›v1å’Œv2çš„æƒå€¼
    public int getWeight(int v1, int v2) {
        return edges[v1][v2];
    }

    // æ˜¾ç¤ºå›¾å¯¹åº”çš„çŸ©é˜µ
    public void show() {
        for (int[] link : edges) {
            System.err.println(Arrays.toString(link));
        }
    }
}

final class MyEntry<K, V> implements Map.Entry<K, V> {
    private final K key;
    private V value;

    public MyEntry(K key, V value) {
        this.key = key;
        this.value = value;
    }

    @Override
    public K getKey() {
        return key;
    }

    @Override
    public V getValue() {
        return value;
    }

    @Override
    public V setValue(V value) {
        V old = this.value;
        this.value = value;
        return old;
    }
}
